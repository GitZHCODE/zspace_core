317  double computeDistance(double &lat1, double &lon1, double &lat2, double &lon2)
318  {
319  double R = 6378.137; // Radius of earth in KM
320 
321  double dLat = (lat2 * PI / 180) - (lat1 * PI / 180);
322  double dLon = (lon2 * PI / 180) - (lon1 * PI / 180);
323 
324  double a = sin(dLat / 2) * sin(dLat / 2) + cos(lat1 * PI / 180) * cos(lat2 * PI / 180) * sin(dLon / 2) * sin(dLon / 2);
325  double c = 2 * atan2(sqrt(a), sqrt(1 - a));
326  double d = R * c;
327 
328  return d; // in kilometers
329  }
