634  zVector inverseKinematics()
635  {
636  // compute target for joint 6
637  zMatrixd Target_J6 = robot_target_matrix * robot_endEffector_matrix;
638 
639  // CALCULATE WRIST CENTER
640  zVector wristC = zVector(Target_J6(0, 3), Target_J6(1, 3), Target_J6(2, 3));
641 
642  wristC -= (zVector(Target_J6(0, 2), Target_J6(1, 2), Target_J6(2, 2)) * Bars[5].linkDH.d * 1);
643 
644  // CALCULATE FIRST 3 ANGLES
645  double th0 = atan2(wristC.y, wristC.x);
646 
647  // x and y of wc in joint 1's basis
648  double r = sqrt(wristC.x * wristC.x + wristC.y * wristC.y) - Bars[0].linkDH.a;
649  double s = Bars[0].linkDH.d - wristC.z;
650 
651  // relevant joint lengths
652  double a1 = Bars[1].linkDH.a;
653  double a2 = sqrt(Bars[2].linkDH.a * Bars[2].linkDH.a + Bars[3].linkDH.d * Bars[3].linkDH.d);
654 
655  // 3rd angle
656  double d = (r * r + s * s - a1 * a1 - a2 * a2) / (2.0 * a1 * a2);
657  double th2 = atan2(sqrt(1.0 - d * d), d); // negate for alt sol'n
658 
659  // 2nd angle
660  double k1 = a1 + a2 * cos(th2);
661  double k2 = a2 * sin(th2);
662  double th1 = atan2(s, r) - atan2(k2, k1);
663 
664 
665  jointRotations[0].rotation = th0 * RAD_TO_DEG;
666  jointRotations[1].rotation = (th1)* RAD_TO_DEG;
667 
668  double dq2 = atan2(Bars[2].linkDH.a, Bars[3].linkDH.d);
669  jointRotations[2].rotation = (((th2 + dq2) * RAD_TO_DEG) - 90 /*- home_rotations[2]*/) * 1;
670 
671 
672 
673  // SET FORWARD
674  forwardKinematics();
675 
676 
677  // SOLVE LAST 3 ANGLES
678 
679  zMatrixd r03 = Bars[0].T * Bars[1].T * Bars[2].T;
680 
681  zMatrixd r36 = r03.transpose() * Target_J6;
682  double t = r36(2, 2);
683 
684  double th3 = atan2(-r36(1, 2), -r36(0, 2));
685  double th4 = atan2(1.0 * sqrt(1.0 - t * t), t);
686  double th5 = atan2(-r36(2, 1), r36(2, 0));
687 
688  jointRotations[3].rotation = (th3 * RAD_TO_DEG);
