48  inline vector<string> splitString(const string& str, const string& delimiter)
49  {
50  vector<string> elements;
51  // Skip delimiters at beginning.
52  string::size_type lastPos = str.find_first_not_of(delimiter, 0);
53  // Find first "non-delimiter".
54  string::size_type pos = str.find_first_of(delimiter, lastPos);
55 
56  while (string::npos != pos || string::npos != lastPos)
57  {
58  // Found a token, add it to the vector.
59  elements.push_back(str.substr(lastPos, pos - lastPos));
60  // Skip delimiters. Note the "not_of"
61  lastPos = str.find_first_not_of(delimiter, pos);
62  // Find next "non-delimiter"
63  pos = str.find_first_of(delimiter, lastPos);
64  }
65  return elements;
66  }
