182  {
183  vector<string> elements;
184  // Skip delimiters at beginning.
185  string::size_type lastPos = str.find_first_not_of(delimiter, 0);
186  // Find first "non-delimiter".
187  string::size_type pos = str.find_first_of(delimiter, lastPos);
188 
189  while (string::npos != pos || string::npos != lastPos)
190  {
191  // Found a token, add it to the vector.
192  elements.push_back(str.substr(lastPos, pos - lastPos));
193  // Skip delimiters. Note the "not_of"
194  lastPos = str.find_first_not_of(delimiter, pos);
195  // Find next "non-delimiter"
196  pos = str.find_first_of(delimiter, lastPos);
197  }
198  return elements;
